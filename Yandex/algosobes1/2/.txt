#include <bits/stdc++.h>
using namespace std;

void erase(int t, int start, string &s)
{
   size_t n = s.size();

   for (int i = start; s[i + t] != " " || i + t < n; i++)
   {
      s[i] = s[i + t];
      //abd  asdfg  aaa
      //abd asdfg  aaa
   }
}

void RemoveSpasec(string &s)
{
   size_t n = s.size();

   int t = 0, count = 0;
   for (int i = 0; i < n; i++)
   {
      if (s[i] == " ")
      {
         t++;
      }
      else
      {
         if (t > 1)
         {
            count += t - 1;
            erase(t - 1, i - t + 2, s);
         }

         t = 0;
      }
   }

   s.resize(n - count + 1);
}

int main()
{
   string s;
   getline(cin , s);

   RemoveSpasec(s);

   cout << s << "\n";
   cout << s.size();

   return 0;
}
